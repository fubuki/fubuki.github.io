### 取得當月的第一天

	$start = date("Y-m-1");	
	$date = new DateTime($start);

### 往前推1個月

	$result = $date->modify( '-1 month' );

### 當月的最後一天

	$result = $date->modify( '+2 month' );
	$result = $date->modify( '-1 day' );
